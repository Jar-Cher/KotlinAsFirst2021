30 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
31     val writer = File(outputName).bufferedWriter()
32     var currentLineLength = 0
33     fun append(word: String) {
34         if (currentLineLength > 0) {
35             if (word.length + currentLineLength >= lineLength) {
36                 writer.newLine()
37                 currentLineLength = 0
38             } else {
39                 writer.write(" ")
40                 currentLineLength++
41             }
42         }
43         writer.write(word)
44         currentLineLength += word.length
45     }
46     for (line in File(inputName).readLines()) {
47         if (line.isEmpty()) {
48             writer.newLine()
49             if (currentLineLength > 0) {
50                 writer.newLine()
51                 currentLineLength = 0
52             }
53             continue
54         }
55         for (word in line.split(Regex("\\s+"))) {
56             append(word)
57         }
58     }
59     writer.close()
60 }

323 fun markdownToHtmlSimple(inputName: String, outputName: String) {
324     File(outputName).bufferedWriter().use {
325         val lines = File(inputName).readLines()
326         var tags = stack<String>()
327         var p = false
328         it.write("<html><body><p>")
329         for ((i, line) in lines.withIndex()) {
330             if (line.isBlank()) continue
331             if (i > 0 && p && lines[i - 1].isBlank()) {
332                 it.write("</p>")
333                 it.newLine()
334                 it.write("<p>")
335             }
336             val htmlLine = StringBuilder("")
337 
338             fun correction(tag: String) {
339                 if (tags.isNotEmpty() && tags.top == tag) {
340                     htmlLine.append(tags.pop().replace("<", "</"))
341                 } else {
342                     tags.push(tag)
343                     htmlLine.append(tag)
344                 }
345             }
346 
347             var i = 0
348             val l = line.length
349             while (i < l) {
350                 p = true
351                 when (line[i]) {
352                     '*' -> {
353                         if (i + 1 < l && line[i + 1] == '*') {
354                             correction("<b>")
355                             i += 2
356                         } else {
357                             correction("<i>")
358                             i += 1
359                         }
360                     }
361                     '~' -> {
362                         if (i + 1 < l && line[i + 1] == '~') {
363                             correction("<s>")
364                             i += 2
365                         }
366                     }
367                     else -> {
368                         htmlLine.append(line[i])
369                         i += 1
370                     }
371                 }
372             }
373             it.write(htmlLine.toString())
374             it.newLine()
375         }
376         it.write("</p></body></html>")
377     }
378 }
